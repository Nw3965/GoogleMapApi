<html>
  <head>
    <meta name="viewport" content="initial-scale=1.0, user-scalable=no">
    <meta charset="UTF-8">

    <title>Fusion Tables Layer Example: Map + Chart</title>

    <link href="/apis/fusiontables/docs/samples/style/default.css"
        rel="stylesheet" type="text/css">
    <script type="text/javascript"
        src="http://maps.google.com/maps/api/js?sensor=false"></script>
    <script type="text/javascript" src="https://www.google.com/jsapi"></script>

    <style type="text/css">
      #chart {
        text-align: center;
        width: 500px;
        padding-top: 10px;
      }
    </style>

    <script type="text/javascript">

      google.load('visualization', '1', {packages: ['corechart']});

      function initialize() {

        var map = new google.maps.Map(document.getElementById('map-canvas'), {
          center: new google.maps.LatLng(37.1, -105),
          zoom: 3,
          mapTypeId: google.maps.MapTypeId.ROADMAP
        });

        var layer = new google.maps.FusionTablesLayer({
          query: {
            select: 'STATE_OR_REGION',
            from: '1zDriu0XmryAOY2N02-I2IbaJCGyvVPRjIgY7_fY'
          },
          options: {
            suppressInfoWindows: true
          },
          map: map
        });

        // Add a listener to the layer that constructs a chart from
        // the data returned on click
        google.maps.event.addListener(layer, 'click', function(e) {

          var data = new google.visualization.DataTable();
          data.addColumn('string', 'Year');
          data.addColumn('number', 'Population');
          var rows = [];
          for (var i = 1910; i <= 2010; i += 10) {
            var year = i.toString();
            var yearValue = parseInt(e.row[year + '_POPULATION'].value, 10);
            rows.push([year, yearValue]);
          }
          data.addRows(rows);

          var chart = new google.visualization.LineChart(
              document.getElementById('chart'));
          var options = {
            title: e.row['STATE_OR_REGION'].value + ' Population',
            height: 400,
            width: 600
          };
          chart.draw(data, options);
        });
      }

      google.maps.event.addDomListener(window, 'load', initialize);

    </script>
  </head>

  <body>
    <div id="map-canvas"></div>
    <div id="chart">Click on a marker to<br>display a chart here</div>
  </body>
